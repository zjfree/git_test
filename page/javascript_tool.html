一句话JS工具

注意：因为浏览器地址栏不允许粘贴 javascript: 开头的字，记得将 j 后的空格去掉。

HTML编辑器

data:text/html,<html contenteditable>

设置当前网页允许编辑

j avascript:document.body.contentEditable=true;

网页变黑白

j avascript:document.body.style.filter='grayscale(1)';0;

网页变反色

j avascript:document.body.style.filter='invert(1)';0;

网页改变色相

j avascript:document.body.style.filter='hue-rotate(180deg)';0;

网页降低亮度

j avascript:document.body.style.filter='brightness(0.8)';0;

网页降低对比度

j avascript:document.body.style.filter='contrast(0.8)';0;

网页模糊

j avascript:document.body.style.filter='blur(3px)';0;

网页降低饱和度

j avascript:document.body.style.filter='saturate(0.5)';0;

仅查看网页文字

j avascript:document.body.innerHTML = document.body.innerText.replace(/\n/ig, "<br>");

查看所有图片

j avascript:h='';for(r of document.querySelectorAll('img')){h+=r.outerHTML+'<br>'};document.body.innerHTML=h;

查看所有链接

j avascript:h='';for(r of document.querySelectorAll('a')){h+=r.outerHTML+'<br>'};document.body.innerHTML=h;

屏蔽所有图片

j avascript:for(r of document.querySelectorAll('img')){r.outerHTML=''}

屏蔽点击

j avascript:for(r of document.querySelectorAll('*')){r.onclick=()=>{return false}}

百度文库提取复制

j avascript:document.body.innerHTML=document.querySelectorAll('#reader-container')[0].outerHTML;

双击隐藏

j avascript:document.body.innerHTML+='<style>div:hover{box-shadow:inset 0 0 10px red;}<style>';for(r of document.querySelectorAll('div')){r.ondblclick=(evt)=>{evt.cancelBubble=true; evt.target.style.visibility='hidden'}}

双击全屏

j avascript:document.body.innerHTML+='<style>div:hover{box-shadow:inset 0 0 10px red;}div:fullscreen,div:-webkit-full-screen{background-color:#fff;}<style>';for(r of document.querySelectorAll('div')){r.ondblclick=(evt)=>{evt.cancelBubble=true; evt.target.requestFullscreen()}}

双击替换

j avascript:document.body.innerHTML+='<style>div:hover{box-shadow:inset 0 0 10px red;}<style>';for(r of document.querySelectorAll('div')){r.ondblclick=(evt)=>{evt.cancelBubble=true; document.body.innerHTML=evt.target.outerHTML}}
